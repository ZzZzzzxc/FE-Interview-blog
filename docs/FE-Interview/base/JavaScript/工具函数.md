# 工具函数

## 防抖

创建一个节流函数，在 wait 秒内最多执行 func 一次的函数。

**极简版**

```js
function debounce(func, wait = 1000) {
  let timerId;
  function debounced() {
    clearTimeout(timerId);
    timerId = setTimeout(func, wait);
  }
  return debounced;
}
```

**lodash 版**

此版本为 lodash 实现的究极简化版，省略了状态控制和外部调用 API ，仅仅实现了最核心的功能，与上述实现不同，lodash 在满足一定条件的时候优先使用 requestAnimationFrame 对 setTimeout 进行替代。

```js
function debounce(func, wait) {
  const root = window;
  let // 参数
    lastArgs,
    // this
    lastThis,
    // 结果
    result,
    // timer
    timerId,
    // 上一次执行debounced函数的时间
    lastCallTime,
    // 上一次调用func的时间
    lastInvokeTime = 0;

  // 如果未设置 wait 且 rAf 可用，则使用 rAf
  const useRAF =
    !wait && wait !== 0 && typeof root.requestAnimationFrame === "function";

  wait = +wait || 0;

  // 执行 func
  function invokeFunc(time) {
    const args = lastArgs;
    const thisArg = lastThis;

    lastArgs = lastThis = undefined;
    lastInvokeTime = time;
    result = func.apply(thisArg, args);
    return result;
  }

  // 创建 timer 
  function startTimer(pendingFunc, wait) {
    if (useRAF) {
      root.cancelAnimationFrame(timerId);
      return root.requestAnimationFrame(pendingFunc);
    }
    return setTimeout(pendingFunc, wait);
  }

  // timer 创建前
  function leadingEdge(time) {
    lastInvokeTime = time;
    timerId = startTimer(timerExpired, wait);
    return result;
  }

  // 计算剩余时间
  function remainingWait(time) {
    // 距离上一次执行 debounced 函数的时间
    const timeSinceLastCall = time - lastCallTime;
    // 剩余等待时间
    const timeWaiting = wait - timeSinceLastCall;
    return timeWaiting;
  }

  // 判断是否需要执行
  function shouldInvoke(time) {
    // 距离上一次执行 debounced 函数的时间
    const timeSinceLastCall = time - lastCallTime;
    // 距离上一次调用 func 的时间
    const timeSinceLastInvoke = time - lastInvokeTime;
    return (
      // 第一次调用
      lastCallTime === undefined ||
      // 距离上一次执行 debounced 函数的时间大于 wait
      timeSinceLastCall >= wait ||
      // 系统时间倒退
      timeSinceLastCall < 0
    );
  }

  // timer 销毁前
  function timerExpired() {
    const time = Date.now();
    // 判断是否需要执行
    if (shouldInvoke(time)) {
      return trailingEdge(time);
    }
    // 重新开始，并通过 remainingWait 计算 wait
    timerId = startTimer(timerExpired, remainingWait(time));
  }

  // timer 销毁
  function trailingEdge(time) {
    timerId = undefined;

    if (lastArgs) {
      return invokeFunc(time);
    }
    lastArgs = lastThis = undefined;
    return result;
  }

  function debounced(...args) {
    const time = Date.now();

    lastArgs = args;
    lastThis = this;
    lastCallTime = time;

    if (shouldInvoke(time)) {
      // 第一次调用
      if (timerId === undefined) {
        // 执行阶段前
        return leadingEdge(lastCallTime);
      }
    }

    // 执行过一次 trailingEdge 后，再执行 debounced 函数，可能会遇到 shouldInvoke 返回 false 的情况，需单独处理
    if (timerId === undefined) {
      timerId = startTimer(timerExpired, wait);
    }

    return result;
  }

  return debounced;
}
```
对于 `lodash` 的实现，我们可以理解为一次 `debounce()` 函数的执行为一个生命周期，生命周期被划分为了以下几个阶段：
- timer 创建前：对应 leadingEdge()，记录 （lastInvokeTime）。
- timer 的创建：对应 startTimer()，如果**未设置 wait 时间且 requestAnimationFrame 可用**，则选用 requestAnimationFrame，否则选用 setTimeout 。
- timer 销毁前：对应 timerExpired()，判断是否要进入下一阶段 ，不需要则重置 timer并回到timer 的创建。
- timer 销毁：对应 trailingEdge()，重置 timer ，执行函数。

## 截取 URL 参数

```js
function getParameterFromUrlByRegExp(url) {
  if (!typeof url === "string") return;
  const res = {};
  url
    .substring(url.indexOf("?") + 1)
    .replace(/([^?&=]+)=([^&]+)/g, (_, key, value) => (res[key] = value));
  return res;
}
```

**正则版**

```js
function getParameterFromUrl(url) {
  if (!typeof url === "string") return;
  const res = {};
  const parameterList = url.substring(url.indexOf("?") + 1).split("&");
  for (const value of parameterList) {
    const tmp = value.split("=");
    res[tmp[0]] = tmp[1];
  }
  return res;
}
```

**lodash版**

```js
const HASH_UNDEFINED = "__lodash_hash_undefined__";

function uniq(array) {
  let index = -1;

  const { length } = array;
  const result = [];

  if (length >= 200) {
    let seen = new Map();
    while (++index < length) {
      let value = array[index];
      value = value !== 0 ? value : 0;
      if (!seen.has(value)) {
        seen.set(value, HASH_UNDEFINED);
        result.push(value);
      }
    }
  } else {
    let seen = result;
    outer: while (++index < length) {
      let value = array[index];
      value = value !== 0 ? value : 0;
      let seenIndex = seen.length;
      while (seenIndex--) {
        if (seen[seenIndex] === value) {
          continue outer;
        }
      }
      result.push(value);
    }
  }
  return result;
}
```
- 对于长度小于 200 的数组，采用两个数组进行存储，一个暂存已经出现的值，一个作为结果返回，两层循环遍历。
- 对于长度大于 200 的数组，暂存数组会改用 `lodash` 内部的 Hash 替代，上述实现则改用 ES6 的 `Map` 进行替代。

## 数组扁平化

### 深度为 1

**loadsh 版**

```js
function flatten(array) {
  const result = [];
  if (array == null) {
    return result;
  }
  for (const value of array) {
    if (Array.isArray(value)) {
      result.push(...value);
    } else {
      result[result.length] = value;
    }
  }
  return result;
}
```

### 深度大于 1

**lodash 版**

```js
function flattenDeep(array, depth = Infinity, result = []) {
  if (array == null) {
    return result;
  }
  for (const value of array) {
    if (depth && Array.isArray(value)) {
      if (depth > 1) {
        flattenDeep(value, depth - 1, result);
      } else {
        result.push(...value);
      }
    } else {
      result[result.length] = value;
    }
  }
  return result;
}
```

**reduce 装x版**

```js
function flattenDeep(arr) {
  return Array.isArray(arr)
    ? arr.reduce((a, b) => [...flattenDeep(a), ...flattenDeep(b)], [])
    : [arr];
}
```
